# Smart message counting
class MessageCounter:
    async def count_message(self, message):
        # Skip if < 3 characters
        if len(message.text or "") < 3:
            return False
        
        # Skip spam detection
        if await self.is_spam(message):
            return False
        
        # Skip flood
        if await self.is_flood(message):
            return False
        
        # Calculate points
        points = self.calculate_points(message)
        
        # Update leaderboard
        await self.update_leaderboard(message.from_user.id, points)
    
    def calculate_points(self, message):
        points = 1  # Base
        
        # Length bonus
        if len(message.text or "") > 50:
            points += 1
        
        # Media bonus
        if message.photo or message.video:
            points += 2
        
        # Voice duration
        if message.voice:
            points += message.voice.duration // 30
        
        return points
    
    async def is_spam(self, message):
        # Check recent messages from user
        recent = await self.get_recent_messages(message.from_user.id, limit=5)
        
        # Same message repeated
        if recent.count(message.text) > 2:
            return True
        
        return False

### Spam Detection for Leaderboard
```python
class SpamDetector:
    def __init__(self):
        self.user_messages = {}  # user_id: [(timestamp, text)]
        self.flood_threshold = 5
        self.flood_window = 10  # seconds
    
    async def is_spam(self, user_id, text):
        now = time.time()
        
        if user_id not in self.user_messages:
            self.user_messages[user_id] = []
        
        # Clean old messages
        self.user_messages[user_id] = [
            (ts, txt) for ts, txt in self.user_messages[user_id]
            if now - ts < 60
        ]
        
        messages = self.user_messages[user_id]
        
        # Check same message repetition
        same_count = sum(1 for _, txt in messages if txt == text)
        if same_count >= 3:
            return True
        
        # Check flood
        recent = [ts for ts, _ in messages if now - ts < self.flood_window]
        if len(recent) >= self.flood_threshold:
            return True
        
        # Check copy-paste pattern
        if len(text) > 100 and self.is_copypasta(messages, text):
            return True
        
        # Add current message
        self.user_messages[user_id].append((now, text))
        
        return False
    
    def is_copypasta(self, messages, current):
        # Check if message is very similar to recent ones
        for _, txt in messages[-5:]:
            similarity = self.similarity_ratio(txt, current)
            if similarity > 0.85:
                return True
        return False
```

    """Global ban user across all groups"""
    if not context.args:
        await update.message.reply_html(
            "<b>✗ Usage Error</b>\n\n"
            "<code>/gban [user] [reason]</code>"
        )
        return
    
    user = await get_user(context.args[0])
    reason = " ".join(context.args[1:]) or "No reason provided"
    
    # Add to global ban list
    await db.add_gban(user.id, reason, update.effective_user.id)
    
    # Ban from all groups
    bot_groups = await db.get_all_groups()
    success = 0
    failed = 0

            await context.bot.ban_chat_member(group.id, user.id)
